import '/flutter_flow/flutter_flow_theme.dart';
import '/flutter_flow/flutter_flow_util.dart';

import 'dart:ui';
import 'package:flutter/material.dart';
import 'package:google_fonts/google_fonts.dart';
import 'package:provider/provider.dart';
import 'dart:math' as math; // 用于角度旋转计算

import 'home_page_model.dart';
export 'home_page_model.dart';

class HomePageWidget extends StatefulWidget {
  const HomePageWidget({super.key});

  static String routeName = 'HomePage';
  static String routePath = '/homePage';

  @override
  State<HomePageWidget> createState() => _HomePageWidgetState();
}

class _HomePageWidgetState extends State<HomePageWidget> {
  late HomePageModel _model;
  final scaffoldKey = GlobalKey<ScaffoldState>();

  @override
  void initState() {
    super.initState();
    _model = createModel(context, () => HomePageModel());
  }

  @override
  void dispose() {
    _model.dispose();
    super.dispose();
  }

  @override
  Widget build(BuildContext context) {
    return GestureDetector(
      onTap: () {
        FocusScope.of(context).unfocus();
        FocusManager.instance.primaryFocus?.unfocus();
      },
      child: Scaffold(
        key: scaffoldKey,
        backgroundColor: Color(0xFFF1F4F8), // 浅灰背景，突显卡片
        appBar: AppBar(
          backgroundColor: FlutterFlowTheme.of(context).primary,
          automaticallyImplyLeading: false,
          title: Text(
            'Helios AI Dashboard',
            style: FlutterFlowTheme.of(context).headlineMedium.override(
                  fontFamily: 'Outfit',
                  color: Colors.white,
                  fontSize: 22,
                  fontWeight: FontWeight.bold,
                ),
          ),
          centerTitle: true,
          elevation: 2,
        ),
        body: SafeArea(
          top: true,
          child: FutureBuilder<ApiCallResponse>(
            future: GetTrackerStatusCall.call(),
            builder: (context, snapshot) {
              // 1. 加载状态 (Loading)
              if (!snapshot.hasData) {
                return Center(
                  child: Column(
                    mainAxisAlignment: MainAxisAlignment.center,
                    children: [
                      CircularProgressIndicator(
                        color: FlutterFlowTheme.of(context).primary,
                      ),
                      SizedBox(height: 10),
                      Text("Connecting to Helios...", style: TextStyle(color: Colors.grey)),
                    ],
                  ),
                );
              }

              // 2. 获取数据 (Get Data)
              final apiResult = snapshot.data!;
              final jsonBody = apiResult.jsonBody;

              // --- 数据解析区 (Safe Parsing) ---
              // 使用辅助函数防止空值报错
              String condition = getJsonField(jsonBody, r'''$.weather_data.condition''')?.toString() ?? "Unknown";
              String suggestion = getJsonField(jsonBody, r'''$.weather_data.suggestion''')?.toString() ?? "No Data";
              String temp = getJsonField(jsonBody, r'''$.sensor_data.temperature''')?.toString() ?? "0";
              
              // 尝试解析数值，如果失败则默认为 0.0
              double humidity = double.tryParse(getJsonField(jsonBody, r'''$.sensor_data.humidity''')?.toString() ?? "") ?? 0.0;
              
              // 光照强度 (取平均值)
              double luxTr = double.tryParse(getJsonField(jsonBody, r'''$.sensor_data.lux_tr''')?.toString() ?? "") ?? 0.0;
              double luxTl = double.tryParse(getJsonField(jsonBody, r'''$.sensor_data.lux_tl''')?.toString() ?? "") ?? 0.0;
              double luxBr = double.tryParse(getJsonField(jsonBody, r'''$.sensor_data.lux_br''')?.toString() ?? "") ?? 0.0;
              double luxBl = double.tryParse(getJsonField(jsonBody, r'''$.sensor_data.lux_bl''')?.toString() ?? "") ?? 0.0;
              double avgLux = (luxTr + luxTl + luxBr + luxBl) / 4; 

              // 舵机角度 (如果 API 还没这字段，默认给 45 度演示)
              double angleX = double.tryParse(getJsonField(jsonBody, r'''$.control.servo_x''')?.toString() ?? "") ?? 45.0;
              double angleY = double.tryParse(getJsonField(jsonBody, r'''$.control.servo_y''')?.toString() ?? "") ?? 30.0;


              // 3. 构建界面 (Build UI)
              return SingleChildScrollView(
                padding: EdgeInsets.all(16),
                child: Column(
                  crossAxisAlignment: CrossAxisAlignment.start,
                  children: [
                    // --- SECTION 1: Weather (AI Result) ---
                    _sectionTitle("Weather & AI Insight"),
                    SizedBox(height: 8),
                    _buildWeatherCard(condition, temp, suggestion),
                    
                    SizedBox(height: 24),
                    
                    // --- SECTION 2: Sensors (Gauges) ---
                    _sectionTitle("Environment Sensors"),
                    SizedBox(height: 8),
                    Row(
                      children: [
                        // 湿度表
                        Expanded(child: _buildCircularGauge("Humidity", humidity, 100, Colors.blueAccent, "%")),
                        SizedBox(width: 12),
                        // 光照表 (假设最大光强 1000)
                        Expanded(child: _buildCircularGauge("Light Intensity", avgLux, 1000, Colors.orangeAccent, "lux")),
                      ],
                    ),

                    SizedBox(height: 24),

                    // --- SECTION 3: Angle Monitor ---
                    _sectionTitle("Solar Tracker Status"),
                    SizedBox(height: 8),
                    _buildAngleMonitor(angleX, angleY),
                    
                    SizedBox(height: 30),
                  ],
                ),
              );
            },
          ),
        ),
      ),
    );
  }

  // --- 辅助小组件 (Widgets) ---

  // 标题样式
  Widget _sectionTitle(String title) {
    return Padding(
      padding: const EdgeInsets.only(left: 4.0),
      child: Text(
        title,
        style: GoogleFonts.inter(
          color: Color(0xFF57636C),
          fontSize: 14,
          fontWeight: FontWeight.w600,
        ),
      ),
    );
  }

  // 1. 天气卡片
  Widget _buildWeatherCard(String condition, String temp, String suggestion) {
    IconData weatherIcon = Icons.wb_sunny;
    Color cardColor = Color(0xFF4B39EF); // 默认 Helios 蓝
    String bgImage = '';

    // 简单的逻辑判断图标和颜色
    String lowerCond = condition.toLowerCase();
    if (lowerCond.contains('cloud')) {
      weatherIcon = Icons.cloud;
      cardColor = Color(0xFF4B39EF).withOpacity(0.8);
    } else if (lowerCond.contains('rain')) {
      weatherIcon = Icons.water_drop;
      cardColor = Color(0xFF39D2C0); // Teal
    } else if (lowerCond.contains('sunny') || lowerCond.contains('clear')) {
      cardColor = Color(0xFFEE8B60); // Orange
    }

    return Container(
      width: double.infinity,
      decoration: BoxDecoration(
        color: cardColor,
        borderRadius: BorderRadius.circular(12),
        boxShadow: [
          BoxShadow(
            blurRadius: 4,
            color: Color(0x33000000),
            offset: Offset(0, 2),
          )
        ],
      ),
      child: Padding(
        padding: EdgeInsets.all(16),
        child: Column(
          crossAxisAlignment: CrossAxisAlignment.start,
          children: [
            Row(
              mainAxisAlignment: MainAxisAlignment.spaceBetween,
              children: [
                Column(
                  crossAxisAlignment: CrossAxisAlignment.start,
                  children: [
                    Text(
                      condition,
                      style: GoogleFonts.outfit(
                        color: Colors.white,
                        fontSize: 24,
                        fontWeight: FontWeight.w500,
                      ),
                    ),
                    Text(
                      'Temperature',
                      style: GoogleFonts.readexPro(
                        color: Color(0xB3FFFFFF),
                        fontSize: 12,
                      ),
                    ),
                  ],
                ),
                Text(
                  '$temp°C',
                  style: GoogleFonts.outfit(
                    color: Colors.white,
                    fontSize: 44,
                    fontWeight: FontWeight.w600,
                  ),
                ),
              ],
            ),
            SizedBox(height: 20),
            Container(
              decoration: BoxDecoration(
                color: Color(0x33FFFFFF),
                borderRadius: BorderRadius.circular(8),
              ),
              padding: EdgeInsets.all(12),
              child: Row(
                children: [
                  Icon(Icons.auto_awesome, color: Colors.white, size: 20),
                  SizedBox(width: 8),
                  Expanded(
                    child: Text(
                      'AI: $suggestion',
                      style: GoogleFonts.readexPro(
                        color: Colors.white,
                        fontSize: 14,
                        fontStyle: FontStyle.italic,
                      ),
                    ),
                  ),
                ],
              ),
            ),
          ],
        ),
      ),
    );
  }

  // 2. 仪表盘 (Gauge)
  Widget _buildCircularGauge(String title, double value, double max, Color color, String unit) {
    double progress = (value / max).clamp(0.0, 1.0);
    
    return Container(
      height: 160,
      decoration: BoxDecoration(
        color: Colors.white,
        borderRadius: BorderRadius.circular(12),
        boxShadow: [
          BoxShadow(blurRadius: 4, color: Color(0x33000000), offset: Offset(0, 2))
        ],
      ),
      child: Padding(
        padding: EdgeInsets.all(12),
        child: Column(
          mainAxisAlignment: MainAxisAlignment.center,
          children: [
            Text(title, style: GoogleFonts.readexPro(color: Color(0xFF57636C), fontSize: 12)),
            SizedBox(height: 12),
            Stack(
              alignment: Alignment.center,
              children: [
                SizedBox(
                  width: 80,
                  height: 80,
                  child: CircularProgressIndicator(
                    value: 1.0, // 背景全圆
                    strokeWidth: 8,
                    color: Color(0xFFE0E3E7),
                  ),
                ),
                SizedBox(
                  width: 80,
                  height: 80,
                  child: CircularProgressIndicator(
                    value: progress,
                    strokeWidth: 8,
                    color: color,
                    strokeCap: StrokeCap.round,
                  ),
                ),
                Column(
                  mainAxisSize: MainAxisSize.min,
                  children: [
                    Text(
                      value.toInt().toString(),
                      style: GoogleFonts.outfit(
                        color: Color(0xFF14181B),
                        fontSize: 20,
                        fontWeight: FontWeight.bold,
                      ),
                    ),
                    Text(
                      unit,
                      style: GoogleFonts.readexPro(
                        color: Color(0xFF57636C),
                        fontSize: 10,
                      ),
                    ),
                  ],
                )
              ],
            ),
          ],
        ),
      ),
    );
  }

  // 3. 角度可视化 (Angle Visualizer)
  Widget _buildAngleMonitor(double x, double y) {
    return Container(
      width: double.infinity,
      padding: EdgeInsets.all(16),
      decoration: BoxDecoration(
        color: Colors.white,
        borderRadius: BorderRadius.circular(12),
        boxShadow: [
          BoxShadow(blurRadius: 4, color: Color(0x33000000), offset: Offset(0, 2))
        ],
      ),
      child: Row(
        mainAxisAlignment: MainAxisAlignment.spaceAround,
        children: [
          _singleAngle("Pan Axis (X)", x, Colors.blue),
          Container(width: 1, height: 60, color: Color(0xFFE0E3E7)),
          _singleAngle("Tilt Axis (Y)", y, Colors.green),
        ],
      ),
    );
  }

  Widget _singleAngle(String label, double angle, Color color) {
    return Column(
      children: [
        // 旋转图标
        Transform.rotate(
          angle: angle * (math.pi / 180), // 将角度转换为弧度
          child: Icon(Icons.navigation, size: 40, color: color),
        ),
        SizedBox(height: 8),
        Text(
          label,
          style: GoogleFonts.readexPro(color: Color(0xFF57636C), fontSize: 12),
        ),
        Text(
          "${angle.toStringAsFixed(1)}°",
          style: GoogleFonts.outfit(
            color: Color(0xFF14181B),
            fontSize: 18,
            fontWeight: FontWeight.bold,
          ),
        ),
      ],
    );
  }
}
